```GDScript
# InteractCap.gd（简化）
extends Node
class_name InteractCap

func enqueue_open_chest(chest : Node):
    var queue : ActionQueue = owner.get_node_or_null("ActionQueue")
    if queue == null:
        queue = ActionQueue.new(owner)
        owner.add_child(queue)

    queue.do(funcref(self, "_block_move"))
    queue.do(funcref(self, "_face_chest", [chest]))
    queue.do(funcref(self, "_play_open_anim"))
    queue.until(owner.get_node("AnimationPlayer"), "animation_finished")
    queue.do(funcref(self, "_give_loot", [chest]))
    queue.wait(0.1)
    queue.do(funcref(self, "_unblock_move"))

func _block_move():     owner.set("can_move", false)
func _unblock_move():   owner.set("can_move", true)
func _face_chest(c):    owner.look_at(c.global_position)
func _play_open_anim(): owner.get_node("AnimationPlayer").play("open")
func _give_loot(c):     c.transfer_loot_to(owner)
```
